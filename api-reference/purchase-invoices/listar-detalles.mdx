---
title: "Listar Detalles"
---

Obtén la lista de todos los detalles asociados a una factura de compra específica.

```http
GET /purchase-invoices/{id}/details
```

<Note>
  Este endpoint requiere el scope **`PurchaseInvoices.Read`**.
</Note>

### Path parameters

<ParamField path="id" type="integer" required>
  Identificador único de la factura de compra.
</ParamField>

### Example request

<CodeGroup>

```bash curl
curl --request GET \
  --url "https://api.primenetcloud.com/v1/purchase-invoices/1054/details" \
  --header "X-Api-Key: YOUR_API_KEY" \
  --header "X-Tenant-Id: TENANT_ID" \
  --header "Content-Type: application/json"
```

```javascript JavaScript
const response = await fetch("https://api.primenetcloud.com/v1/purchase-invoices/1054/details", {
  method: "GET",
  headers: {
    "X-Api-Key": "YOUR_API_KEY",
    "X-Tenant-Id": "TENANT_ID",
    "Content-Type": "application/json",
  },
});

const data = await response.json();
console.log(data);
```

```python Python
import requests

response = requests.get(
    "https://api.primenetcloud.com/v1/purchase-invoices/1054/details",
    headers={
        "X-Api-Key": "YOUR_API_KEY",
        "X-Tenant-Id": "TENANT_ID",
    },
)

print(response.json())
```

</CodeGroup>

### Response fields

<ResponseField name="id" type="integer">
  Identificador único del detalle.
</ResponseField>

<ResponseField name="itemType" type="object">
  Tipo de ítem.
  <Expandable title="Ver propiedades">
    <ResponseField name="id" type="string">
      Identificador del tipo de ítem.

      Valores posibles:

      - `Charge`
      - `FixedAsset`
      - `Product`
      - `ChartOfAccount`
    </ResponseField>
    <ResponseField name="name" type="string">
      Nombre del tipo de ítem traducido.
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="item" type="object">
  Información del ítem asociado al detalle.
  <Expandable title="Ver propiedades">
    <ResponseField name="id" type="integer">
      Identificador del ítem.
    </ResponseField>
    <ResponseField name="name" type="string">
      Nombre del ítem.
    </ResponseField>
    <ResponseField name="code" type="string">
      Código del ítem (aplica para cuentas contables).
    </ResponseField>
    <ResponseField name="fastCode" type="string">
      Código rápido (aplica para cargos).
    </ResponseField>
    <ResponseField name="isTransparent" type="boolean">
      Indica si el cargo es transparente (aplica para cargos).
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="taxPercent" type="decimal">
  Porcentaje de impuesto aplicado.
</ResponseField>

<ResponseField name="acquiredWarranty" type="string">
  Fecha de adquisición de la garantía en formato ISO 8601 (aplica para activos fijos).
</ResponseField>

<ResponseField name="description" type="string">
  Descripción del detalle.
</ResponseField>

<ResponseField name="calculationType" type="object">
  Tipo de cálculo.
  <Expandable title="Ver propiedades">
    <ResponseField name="id" type="string">
      Identificador del tipo de cálculo.
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="quantity" type="decimal">
  Cantidad del ítem.
</ResponseField>

<ResponseField name="productPackaging" type="object">
  Empaque del producto.
  <Expandable title="Ver propiedades">
    <ResponseField name="name" type="string">
      Nombre del empaque.
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="packagingQuantity" type="decimal">
  Cantidad de empaques.
</ResponseField>

<ResponseField name="price" type="decimal">
  Precio unitario.
</ResponseField>

<ResponseField name="discount" type="decimal">
  Descuento aplicado.
</ResponseField>

<ResponseField name="subTotalAmount" type="decimal">
  Monto subtotal calculado.
</ResponseField>

<ResponseField name="taxAmount" type="decimal">
  Monto total de impuestos calculados.
</ResponseField>

<ResponseField name="totalAmount" type="decimal">
  Monto total calculado (Subtotal - Descuento + Impuestos).
</ResponseField>

<ResponseField name="unitReceived" type="decimal">
  Unidades recibidas.
</ResponseField>

<ResponseField name="operation" type="object">

      Identificador de la operación.
    </ResponseField>
    <ResponseField name="transactionNumber" type="integer">
      Número de transacción de la operación.
    </ResponseField>
    <ResponseField name="category" type="string">
      Categoría de la operación logística.

      Valores posibles:

      - `Shipment`
      - `CustomManagement`
      - `Additional`
      - `InlandFreight`
    </ResponseField>
    <ResponseField name="movementType" type="object">
      Tipo de movimiento (solo aplica para embarques).
      <Expandable title="Ver propiedades">
        <ResponseField name="id" type="string">
          Identificador del tipo.

          Valores posibles:

          - `Import`
          - `Export`
          - `Transshipment`
          - `Crosstrade`
          - `Transit`
        </ResponseField>
        <ResponseField name="name" type="string">Nombre del tipo.</ResponseField>
      </Expandable>
    </ResponseField>
    <ResponseField name="transportType" type="object">
      Tipo de transporte (solo aplica para embarques).
      <Expandable title="Ver propiedades">
        <ResponseField name="id" type="string">
          Identificador del tipo.

          Valores posibles:

          - `Air`
          - `Sea`
          - `Ground`
          - `Rail`
        </ResponseField>
        <ResponseField name="name" type="string">Nombre del tipo.</ResponseField>
      </Expandable>
    </ResponseField>
    <ResponseField name="bl" type="string">
      Número de Bill of Lading (solo aplica para embarques).
    </ResponseField>
    <ResponseField name="customer" type="object">
      Cliente asociado a la operación (solo aplica para embarques).
      <Expandable title="Ver propiedades">
        <ResponseField name="name" type="string">Nombre del cliente.</ResponseField>
      </Expandable>
    </ResponseField>
    <ResponseField name="master" type="object">
      Master asociado (solo aplica para embarques).

        <ResponseField name="id" type="integer">Identificador del master.</ResponseField>
        <ResponseField name="transactionNumber" type="integer">Número de transacción.</ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="project" type="object">
